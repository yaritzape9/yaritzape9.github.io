	def pretty_board(board_string)
	  string = board_string.chars
	  board = []
	  9.times do |index|
	    board << string.shift(9)
	  end
	    board

	  9.times do |row|
	    if row % 3==  0
	      puts "+-----+-----+-----+"
	    end
	  9.times do |col|
	    if col == 0
	      print "|" + board[row][col] +" "
	    elsif col == 2 || col == 5
	      print board[row][col]+"|"
	    elsif col == 8
	      puts board[row][col]+"|"
	    else
	      print board[row][col]+" "
	    end
	   end
	  end
	 puts "+-----+-----+-----+"
       end
